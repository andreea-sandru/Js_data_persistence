CREATE TABLE PERSON(
CNP VARCHAR(255), 
birthDate DATE DEFAULT SYSDATE, 
name VARCHAR(255), 
gender VARCHAR(255),
CONSTRAINT PK1 PRIMARY KEY(CNP)
);

CREATE TABLE EMPLOYEE(
CNP VARCHAR(255), 
birthDate DATE DEFAULT SYSDATE, 
name VARCHAR(255), 
gender VARCHAR(255),
job VARCHAR(255),
CONSTRAINT PK2 PRIMARY KEY(CNP)
);

CREATE TABLE TEACHER(
CNP VARCHAR(255), 
birthDate DATE DEFAULT SYSDATE, 
name VARCHAR(255), 
gender VARCHAR(255),
job VARCHAR(255),
subject VARCHAR(255), 
school VARCHAR(255),
CONSTRAINT PK3 PRIMARY KEY(CNP)
);

CREATE TABLE DepartmentHead(
CNP VARCHAR(255), 
birthDate DATE DEFAULT SYSDATE, 
name VARCHAR(255), 
gender VARCHAR(255),
job VARCHAR(255),
subject VARCHAR(255), 
school VARCHAR(255),
bonus NUMBER(7),
CONSTRAINT PK4 PRIMARY KEY(CNP)
);

CREATE TABLE Administrator(
CNP VARCHAR(255), 
birthDate DATE DEFAULT SYSDATE, 
name VARCHAR(255), 
gender VARCHAR(255),
job VARCHAR(255),
institution VARCHAR(255),
CONSTRAINT PK5 PRIMARY KEY(CNP)
);

CREATE TABLE PRODEAN(
CNP VARCHAR(255), 
birthDate DATE DEFAULT SYSDATE, 
name VARCHAR(255), 
gender VARCHAR(255),
job VARCHAR(255),
college VARCHAR(255),
CONSTRAINT PK6 PRIMARY KEY(CNP)
);

CREATE TABLE DEAN(
CNP VARCHAR(255), 
birthDate DATE DEFAULT SYSDATE, 
name VARCHAR(255), 
gender VARCHAR(255),
job VARCHAR(255),
college VARCHAR(255),
subordinate VARCHAR(255),
CONSTRAINT PK7 PRIMARY KEY(CNP)
);

CREATE TABLE STUDENT(
CNP VARCHAR(255), 
birthDate DATE DEFAULT SYSDATE, 
name VARCHAR(255), 
gender VARCHAR(255),
college VARCHAR(255),
grades VARCHAR(255),
CONSTRAINT PK8 PRIMARY KEY(CNP)
);
